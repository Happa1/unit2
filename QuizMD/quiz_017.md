def get_l3tt3r (msg:str)->str:
    output=""
    for i in msg:
        if i=="a":
            output+="4"
        elif i=="e":
            output+="3"
        elif i=="i":
            output+="1"
        elif i=="o":
            output+="0"
        elif i==" ":
            output+="_"
        else:
            output+=i

    return output

case1=get_l3tt3r(msg="Hello World")
